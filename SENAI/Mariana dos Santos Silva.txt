,atual
,COUNT(department_id) qtd_departamentos
FROM employees
INNER JOIN employee_department_history ON employees.employee_id = employee_department_history.employee_id
INNER JOIN (select employee_id,
	case 
	when end_date is null then department_name
else
	null
end atual
from employee_department_history
INNER JOIN departments ON departments.department_id = employee_department_history.department_id
where end_date is null) departamento ON departamento.employee_id = employee_department_history.employee_id
group by employee_name,atual
